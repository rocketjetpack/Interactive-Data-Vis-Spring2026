## Examples

1. Create a plot
<!--```js minimal_baseline

    marks: [
        Plot.frame(),
    ]
})
```-->

```js
const text_to_render = "Hi There!";
display(Plot.plot({
    title: "This is a Frame",
    subtitle: "This is the subtitle.",
    marks: [
        Plot.frame(),
        Plot.text([text_to_render], { frameAnchor: "middle"})
    ]
}))
```

2. Define some data
```js
const weather_data = [
    { date: "Mar 7", temperature: 68 },
    { date: "Mar 8", temperature: 45 },
    { date: "Mar 9", temperature: 67 },
    { date: "Mar 10", temperature: 74 },
    { date: "Mar 11", temperature: 81 }
]
```

3.Plot it
```js
display(Plot.plot({
    title: "Temperature vs Date",
    subtitle: "Some temperature data",
    grid: true,
    y: {
        domain: [0, 90]
    },
    x: { 
        domain: ["Mar 7", "Mar 8", "Mar 9", "Mar 10", "Mar 11"]
    },
    marks: [
        Plot.frame(),
        Plot.barY(
            weather_data,
            { x: "date", y: "temperature", fill: "blue", stroke: "black"}
        ),
        Plot.dot(
            weather_data,
            { x: "date", y: (d) => d.temperature, fill: "pink", r: 10, stroke: "black" }, // Use function based y so that ops could be done like +5
        ),
        Plot.line(
            weather_data,
            { x: "date", y: "temperature"}
        ),
    ]
}))
```
